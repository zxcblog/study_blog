# etcdctl 命令行命令
```shell
# 查看命令
$ etcdctl -h

# 查看etcd成员列表
$ etcdctl --endpoints=localhost:2379 member list
1b213fb4df774056, started, etcd1, http://172.26.0.101:2380, http://172.26.0.101:2379, false
d61a3fd1a7c242c9, started, etcd3, http://172.26.0.103:2380, http://172.26.0.103:2379, false
d67978f05cd470e5, started, etcd2, http://172.26.0.102:2380, http://172.26.0.102:2379, false

# 向etcd集群中写入数据
$ etcdctl put test 1

# 查看etcd集群中的数据
$ etcdctl get test
test
1

# 以16进制进行返回
$ etcdctl get test --hex
\x74\x65\x73\x74
\x31

# 获取指定前缀的值， --limit 限制返回的key数量
$ etcdctl get --prefix t --limit 3
test
1

# 按key的字典顺序读取
# 读取字典顺序大于或等于test的key
$ etcdctl get --from-key test

# etcd可以通过读取一个key来获取当前etcd服务端的版本号，不管key是否存在
# cluster_id: 请求的etcd集群id, member_id: 请求的etcd节点ID
# revision: etcd服务端当前全局数据版本号。对任一key的put或delete操作都会使revision自增1， revision=1是etcd的保留版本号， 因此用户的key版本号将从2开始
# raft_term: etcd当前raft主节点任期号
# create_revision: 当前key创建时全局数据版本号revision值
# mod_revision: 当前key最后一次修改时全局数据版本号revision的值
# version: 当前key的数据版本号。key创建时version为1，对当前key进行put操作会使version自增1，将key删除后，重新创建version又会从1开始计数
$ etcdctl get test -w=json
{"header":{"cluster_id":5722707360917884858,"member_id":15427713643206296265,"revision":2,"raft_term":2},"kvs":[{"key":"dGVzdA==","create_revision":2,"mod_revision":2,"version":1,"value":"MQ=="}],"count":1}

# 访问以前版本的key
$ etcdctl get --rev=2 test
test
1


# 删除
$ etcdctl del test
1

# 返回删除成功的键值
$ etcdctl del test --prev-key
1
test 
1

# watch 监测一个键值的变化， 一旦键值发生更新，就会输出最新的值
$ etcdctl watch test

# 租约(lease) 类似redis的ttl, etcd中的键值对可以绑定到租约上，实现存活周期控制，应用客户端可以为etcd集群里面的键授予租约
# 一旦租约的ttl到期， 租约就会过期并且所有附带的键都将被删除
$ etcdctl lease grant 30 # 授予租约 ttl 为30s
lease 42c98a6eb1d3e90c granted with TTL(30s)

# 附件键到租约
$ etcdctl put --lease=42c98a6eb1d3e90c test 121

# 撤销租约 通过租约ID可以撤销租约。撤销租约将删除所有附带的key
$ etcdctl lease revoke 42c98a6eb1d3e90c

# 查看租约是否存活
$ etcdctl lease keep-alive 42c98a6eb1d3e90c

# 查看租约过期时间
$ etcdctl lease timetolive 42c98a6eb1d3e90c

# 查看租约过期时间和绑定到租约上的键
$ etcdctl lease timetolive --keys 42c98a6eb1d3e90c
```

# etcd 权限管理
## 用户管理
```shell
# 查看是否开启权限
$ etcdctl auth status

# 开启权限控制, 开启权限控制需要创建root用户
$ etcdctl auth enable

# 创建用户
$ etcdctl user add root

# 删除用户
$ etcdctl user del root

# 修改密码
$ etcdctl user password root

# 查看所有用户
$ etcdctl user list

# 查看指定用户以及绑定角色
$ etcdctl user get root  
```

## 角色管理
```shell
# 创建角色
$ etcdctl role add root

# 给角色赋权 etcdctl grant-permission role_name [read|write|readwrite] test
# 给admin角色赋予 / 节点的读写权限
$ etcdctl role grant-permission admin readwrite /

# 用户关联角色
$ etcdctl user grant-role admin

# 回收角色赋权
$ etcdctl role revoke-permission role_name test

# 集群权限管理
$ etcdctl --endpoints http://127.0.0.1:2379,http://127.0.0.1:2379,http://127.0.0.1:2379 user add root

# 开启集群认证
$ etcdctl --endpoints http://127.0.0.1:2379,http://127.0.0.1:2379,http://127.0.0.1:2379 auth enable
```
